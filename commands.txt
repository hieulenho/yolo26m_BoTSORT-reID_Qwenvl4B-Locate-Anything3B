$SourceVideo = "F:\videos\1.mp4"

if (-not (Test-Path -LiteralPath $Tracks)) {

}

# Recommended: reuse existing tracking and run all semantic pipelines.

  -SourceVideo $SourceVideo `
  -Tracks $Tracks `
  -SkipTracking `
  -Query $Query `
  -Pipelines A,B,C `
  -LocateBackend locate_anything `
  -RunQwenModel `
  -Device cuda `
  -TorchDtype auto `
  -Quantization 8bit `
  -MaxKeyframes 2 `
  -MaxTracks 20 `
  -MaxCropsPerTrack 1 `
  -MaxNewTokens 512 `
  -LocateMaxFrames 6 `
